HTML5 Tag Examples

<!DOCTYPE html>
<html>
    <head>
        <title>Title</title>
        <meta charset="UTF-8" />
        <meta name="author" content="author's name" />
        <meta http-equiv="Content-Type" content="text/html" />
        <link rel="stylesheet" href="stylesheet.css" type="text/css" media="screen" />
        <base target="_self" /> 
        <base href="http://www.example.com" />
        <style type="text/css" media="screen">
            /* style rules */
        </style>
        <script type="text/javascript" src="external.js" async></script>
        <script type="text/javascript">
            // js content
        </script>
    </head>
    <body>
        <!-- Logical Sections, sectioning content model -->
        <article><!-- content meant to be excerpted as a whole -->
            <header><!-- introductory content -->
                <hgroup><!-- puts headers together for TOC purposes -->
                    <h1><!-- Main heading --></h1>
                    <h2><!-- Sub heading --></h2>
                    <h3><!-- Sub heading --></h3>
                    <h4><!-- Sub heading --></h4>
                    <h5><!-- Sub heading --></h5>
                    <h6><!-- Sub heading --></h6>
                </hgroup>                
                <nav>
                    <!-- navigation information -->
                </nav>
            </header>
            <section>
                <!-- thematically related content -->                
            </section>
            <aside>
                <!-- tangential, supporting content -->
            </aside>
            <footer>
                <!-- author info, licensing, supplemental nav -->
            </footer>
        </article>
        
        <!-- Meaningful tags for content markup, sectioning content model -->
        <p><!-- paragraph content --></p>
        
        <blockquote cite="http://www.example.com">
            <p><!-- quoted content inside block level children --></p>
        </blockquote>
        
        <address>
            <!-- Byline / attribution for those responsible for content -->
        </address>
        
        <pre>
            <!-- preformatted text, preserves whitespace -->
        </pre>
        
        <hr /><!-- horizontal rule, void, replaced element -->
        
        <div>
            <!-- semantically neutral, general purpose content division -->
        </div>
        
        <figure>
            <!-- image, video, quotation, code listing, or other content -->
            <!-- self contained, may be referenced from content as a single unit -->
            <figcaption>
                <!-- Caption or legend for other contents of parent figure element -->
            </figcaption>
        </figure>
        
        <iframe src="otherpage.html" height="100" width="100">
            <!-- Inline / floating frame, displaying separate content -->
        </iframe>
        
        <!-- List content, sectioning content model -->
        <ul>
            <li><!-- list item content --></li>
        </ul>
        
        <ol type="a">
            <li><!-- list item content --></li>
        </ol>
        
        <dl>
            <dt><!-- term --></dt>
            <dd><!-- data --></dd>
        </dl>
        
        <!-- Phrasing Elements -->
        
        <em><!-- emphasized content --></em>
        <strong><!-- strongly emphasized content --></em>
        <time datetime="2012-05-05T15:35:42-07:00">Saturday, May 5, 2012, 3:35 pm Pacific</time>
        <mark><!-- marks a passage of interest within a section --></mark>
        <cite><!-- Name of source, other citation info --></cite>
        <q><!-- short, inline quotation --></q>
        <dfn><!-- The defining instance of a term, especially one that reoccurs --></dfn>
        <abbr title="The Expanded Form"><!-- Shortened form of a word or phrase --></abbr>
        <small><!-- Side note or fine print --></small>
        <span><!-- Generic phrase element --></small>
        <code><!-- computer code --></code>
        <kbd><!-- text or commands for the user to enter --></kbd>
        <samp><!-- sample output of a program or script --></samp>
        <var><!-- programming variable or argument --></var>
        <del cite="url_of_change_info" datetime="change time"><!-- deleted text --></del>
        <ins cite="url_of_change_info" datetime="change time"><!-- inserted text --></ins>
        
        <!-- Offset Text -->
        <i><!-- word or phrase in an alternate voice --></i>
        <b><!-- calls attention to a word or phrase --></b>
        <u><!-- underlines text, marks it visually --></u>
        <s><!-- text that is no longer accurate or relevant --></s>
        
        <sup><!-- superscripted --></sup>
        <sub><!-- subscripted --></sub>
        
        <!-- Text Annotations -->
        <ruby><!-- pronunciation guide in logographic language --></ruby>
        <rp><!-- word or phrase with annotation --></rp>
        <rt><!-- Actual annotation text --></rt>
        
        <!-- Bidirectional Text -->
        <bdi><!-- Bidirectional isolate, isolates a piece of text --></bdi>
        <bdo dir="rtl"><!-- Bidirectional override, explicitly reverses from main --></bdo>
        
        <!-- Line Breaks -->
        <br /><!-- Explicit line break -->
        <wbr /><!-- Line break opportunity -->
    </body>
</html>
