Datatypes: char, int, float, double
Modifiers: short, long, signed, unsigned

Constants/Literals:

  int: 1      long int: 1L      unsigned int: 1U    unsigned long: 1UL
  Octal int:  01        Hex int: 0x1
  float: 1.1f or 1.1e-2f
  double: 1.1
  character: 'x'  ← value is numeric
  byte: '\000' or '\xhh'

Escapes: \a \b \f \n \r \t \v \\ \? \' \" \000

String Literals: "this is a string" ← array of characters, terminated by '\0' null character

Enumerations: 

  enum boolean { NO, YES } ← 0 and 1, respectively
  enum months  { JAN = 'one' }

Declarations:

  type nameA, nameB, nameC;
  type name = initialValue;
  const type name; ← won't change
  register type name; ← mutable, used frequently

Operators:

  + - * / % > >= < <= == != ++ --
  & | ^ << >> ~
  = += -= *= /= %= <<= >>= &= ^= |=

Control flow:

  { naked block }
  if (condition) { block }
  if (condition) { block } else { block }
  if (condition) { block } else if (condition) { block } else { block }
  switch (expression) {
    case const-expression: statements
  }
  while (expression) { block }
  for (ex1; ex2; ex3) { block }
  do statement while (condition);
  while (1) { ... break;    ... }
  while (1) { ... continue; ... }

  for (...)
    for (...)
      if (x)
        goto error;

  error: statement

Functions: returnType name (args) { ... return 0; }

Preprocessor statements:

  #include filename
  #include <filename>
  #define name replacementText
  #define max(A,B) ((A) > (B) ? (A) : (B))
  #undef somename
  #if int boolean
  #elif int boolean
  #endif
  #ifdef X
  #ifndef X

Pointers:

  int x = 1;
  int *ip;    ← declares pointer type variable
  ip = &x;    ← takes pointer to x
  *ip = 2;    ← dereferences and sets x
  int *ipp;   ← declares second pointer
  ipp = ip;   ← sets value of ipp to pointer in ip
  *ipp = 3;   ← sets x

Pointers and Function Args

  /* All args are passed by value, so pointers must be passed in. */
  void swap(int *px, int *py) { /* interchange values by pointer */
    int temp;
    temp = *px;
    *px = *py;
    *py = temp;
  }

Pointers and Arrays

  int x, a[10];    ← declare int array of size 10
  int *pa;         ← establishes pointer
  pa = &a[0];      ← addresses a[0], equivalent to a itself
  x = *pa;         ← sets x to value in a[0]
  x = *(pa+1);     ← sets x to a[1]

  /* get length of string via array pointer addition */
  int strlen(char *s) {
    int n;
    for (n = 0; *s != '\0'; s++) 
      n++;
    return n;
  }
