CoffeeScript Cheatsheet

Syntax

General
  CS files must be pure coffeescript.
  No semi-colons allowed.
  White space is significant.
  Braces are replaced with a tab.

Comments
  Single line comments start with a hash (#).
  Multiline comments are enclosed in triple hashes (###).

Scope
  All scripts are wrapped in an anonymous function, creating a local scope.
  The 'var' keyword is prepended to all variable declarations.
  You can explicitly set globals on the window object.

Functions

myFunc1 = -> "somestring"          # oneliner function
myFunc2 = ->                       # block function
  "somestring"

times = (a,b) -> a * b             # declaration with args
times = (a = 1, b = 2) -> a * b    # args with defaults

sum = (nums...) ->                 # splat for multiargs
  result = 0
  nums.foreach (n) -> result += n
  result

sum([1,2,3])                       # Invocation with parens
sum.apply([1,2,3])                 # Invocation with apply()
sum.call([1,2,3])                  # Invocation with call()

Objects and Arrays

object1 = {one: 1, two: 2}            # Using JSON
object2 = one: 1, two: 2              # Without optional braces
object3 =                             # Commas replaced with newlines
  one: 1
  two: 2

User.create(name: "John Smith")

Flow Control

if true == true                       # Basic if
  "True"

if true != true then "False"          # Oneline if with 'then'

alert "True" if value == 5            # Suffixed if

if ! true then "False"                # ! and 'not' are equivalent
if not true then "False"

unless true                           # Basic unless
  "False"

if true is 1                          # 'is' gives '==='
  "Type coercion fail."

if true isnt true                     # 'isnt' gives 'is not'
  alert "Opposites!"

Loops and Comprehensions

for name in ["Roger", "Roderick", "Brian"]          # Foreach loop
  alert "Release #{name}"

for name, i in ["Roger", "Roderick"]                # Foreach with index
  alert "#{i} - Release #{name}"

release p for p in ["Roger", "Roderick"]            # Online postfix foreach

prisoners = ["Roger", "Roderick", "Brian"]          
release p for p in prisoners when p[0] is "R"       # Filtered comprehension

names = sam: seaborn, donna: moss
alert("#{first} #{last}") for first, last of names  # Comprehension for object

num = 6
minstrel = while num -= 1                           # While acts like map()
  num + " Brave Sir Robin ran away"                 # and returns an array

Arrays

array1 = [1,2,3]                                    # Using JS array syntax
array2 = [                                          # Commas replaced with newlines
  1
  2
  3
]
array3 = [1,2,3,]                                   # Trailing comma stripped by CS

range = [1..5]                                      # Range expands to array

firstTwo = ["one","two","three"][0..1]              # Range after array calls slice()

numbers = [0..9]                        
numbers[3..5] = [-3,-4,-5]                          # Can be used to replace slice

my = "my string"[0..1]                              # Ranges usable on strings

words = ["rattled","rowdy","rebbles","ranks"]
alert "Stop" if "ranks" in words                    # Using 'in' for array search
