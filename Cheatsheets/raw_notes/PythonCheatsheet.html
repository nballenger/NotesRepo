Conditionals

if/elif/else

var1 = 100
if var1:
    print "It's true."
    
if var1:
    print "It's true."
else:
    print "Not true."
    
if var1 == 100:
    print "Was 100."
elif var1 == 101:
    print "Was 101."
elif var1 == 102:
    print "Was 102."
else:
    print "Was something else."

Loops

While

count = 0
while (count < 9):
    print 'Count is: ', count
    count = count + 1

For

for letter in 'Python':
    print 'Current letter: ', letter

fruits = ['banana', 'apple', 'mango']
for fruit in fruits:
    print 'Current fruit: ', fruit

for index in range(len(fruits)):
    print 'Current: ', fruits[index]
    
for num in range(10,20):
    for i in range(2,num):
        if num % i == 0:
            j = num/1
            print '%d equals %d * %d' % (num, i, j)
            break
    else:
        print num, ' is prime'

Break/Continue/Pass

var = 10
while var > 0:
    print 'Current: ', var
    var = var - 1
    if var == 5:
        break

var = 10        
while var > 0:
    var = var - 1
    if var == 5:
        continue
    print 'Current: ', var
    
for letter in 'Python':
    if letter == 'h'
        pass
        print 'This is pass block'
    print 'Current: ', letter
